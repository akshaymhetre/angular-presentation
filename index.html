SVG vs other image types


  • Vector logic in XML format
  • Retains image quality on resized or zoomed
  • Relatively small in size
  • Not supported in some browsers


PNG     SVG

My Directive



					
					


<script>
var myApp = angular.module('MyApp', []);
myApp.directive("svgSrc", function() {
    return {
        restrict: 'A',  // 'A' 'E' C' 'AEC'
        link: function(scope, element, attr) { 
            var fallBack = attr.fallBack || 'png';
            var isSvgSupported = Modernizr.svg;
            var newImgSrcVal = 
                attr.svgSrc + "." + (isSvgSupported ? 'svg' : fallBack); 
            element.attr('src', newImgSrcVal);
        },
        replace: false
    };
});
</script>
